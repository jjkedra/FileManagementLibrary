270 * If client used libfs_notify, then the daemon will send a notification to the client if the file was modified.

272void Daemon::send_notification(char *fname) {
273 std::vector<pid_t> owners;
274
275 for (const auto& handle : this->fileHandles.getHandles()) {
276 if ((handle != nullptr) && (handle->flags & NOTIFY_FLAG) && (std::strcmp(handle->name, fname) == 0)) {
277 owners.push_back(handle->owner);
278 }
279 }
280
281 for (pid_t owner : owners) {
282 // Send the SIGUSR1 signal to each owner
283 kill(owner, SIGUSR1);
284 }
285}
